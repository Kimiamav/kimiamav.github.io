package com.scor.peopledoc.web.gui;

import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.nio.file.StandardCopyOption;

import javax.servlet.annotation.MultipartConfig;
import javax.servlet.http.HttpSession;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.multipart.MultipartFile;




import com.scor.peopledoc.config.ApplicationProperties;

import lombok.AllArgsConstructor;
import lombok.extern.slf4j.Slf4j;


@Controller
@RequestMapping("/hr")
@Slf4j
@AllArgsConstructor(onConstructor_ = { @Autowired })
@MultipartConfig(fileSizeThreshold = 1024 * 1024 * 1, // 1 MB
		maxFileSize = 1024 * 1024 * 10, // 10 MB
		maxRequestSize = 1024 * 1024 * 15 // 15 MB
)
public class DefaultController {


	private final ApplicationProperties applicationProperties;

	@GetMapping("/uploadfile")
	public String getUploadFile(Model mdoel) {
		return "hr/uploadfile";
	}

	@PostMapping("/upload")
	public String postUpload(@RequestParam("file") MultipartFile file,@RequestParam("country") String country,Model model,HttpSession  session) {
		log.debug("method uploadFile");

		try {
			// Exemple : sauvegarde en local
			Path path = Paths.get(applicationProperties.getRootPath() + "/payslips/"+country+"/"+ file.getOriginalFilename());
			Files.copy(file.getInputStream(), path, StandardCopyOption.REPLACE_EXISTING);
			session.setAttribute("fileName", file.getOriginalFilename());
            session.setAttribute("country",country);

			model.addAttribute("message", "Upload OK : " + file.getOriginalFilename());
            model.addAttribute("country",country);
            model.addAttribute("fileName",file.getOriginalFilename());
		} catch (Exception e) {
			model.addAttribute("message", "Erreur durant l’upload : " + e.getMessage());
		}
		return "hr/result";
	}

	@GetMapping("/result")
	public String getResult(Model model,HttpSession  session) {
		model.addAttribute("fileName",session.getAttribute("fileName"));
        model.addAttribute("country",session.getAttribute("country"));
		return "hr/result";
	}

}
